106 T* chkpnt_soa2aos(T* data, int cnt, int sz, int layout, int* permute) {
107  // inverse of F -> data. Just a copy if layout=1. If SoA,
108  // original file order depends on padding and permutation.
109  // Good for a, b, area, v, diam, Memb_list.data, or anywhere values do not change.
110  T* d = new T[cnt * sz];
111  if (layout == 1) { /* AoS */
112  for (int i = 0; i < cnt * sz; ++i) {
113  d[i] = data[i];
114  }
115  } else if (layout == 0) { /* SoA */
116  int align_cnt = nrn_soa_padded_size(cnt, layout);
117  for (int i = 0; i < cnt; ++i) {
118  int ip = i;
119  if (permute) {
120  ip = permute[i];
121  }
122  for (int j = 0; j < sz; ++j) {
123  d[i * sz + j] = data[ip + j * align_cnt];
124  }
125  }
126  }
127  return d;
128 }

827 void checkpoint_restore_tqueue(NrnThread& nt, const Phase2& p2) {
828  int type;
829  checkpoint_restored_ = true;
830 
831  for (int i = 0; i < nt.n_vecplay; ++i) {
832  VecPlayContinuous* vpc = (VecPlayContinuous*)nt._vecplay[i];
833  auto& vec = p2.vec_play_continuous[i];
834  vpc->last_index_ = vec.last_index;
835  vpc->discon_index_ = vec.discon_index;
836  vpc->ubound_index_ = vec.ubound_index;
837  }
838 
839  // PatternStim
840  patstim_index = p2.patstim_index; // PatternStim
841  if (nt.id == 0) {
842  patstim_te = -1.0; // changed if relevant SelfEvent;
843  }
844 
845  for (int i = 0; i < nt.n_presyn; ++i) {
846  nt.presyns_helper[i].flag_ = p2.preSynConditionEventFlags[i];
847  }
848 
849  for (const auto& event: p2.events) {
850  checkpoint_restore_tqitem(event.first, event.second, nt);

854 // A call to finitialize must be avoided after restoring the checkpoint
855 // as that would change all states to a voltage clamp initialization.
856 // Nevertheless t and some spike exchange and other computer state needs to
857 // be initialized.
858 // Also it is occasionally the case that nrn_init allocates data so we
859 // need to call it but avoid the internal call to initmodel.
860 // Consult finitialize.c to help decide what should be here
861 bool checkpoint_initialize() {

864  nrn_spike_exchange_init();
865 
866  // in case some nrn_init allocate data we need to do that but do not
867  // want to call initmodel.
868  _nrn_skip_initmodel = true;
869  for (int i = 0; i < nrn_nthread; ++i) { // should be parallel
870  NrnThread& nt = nrn_threads[i];
871  for (NrnThreadMembList* tml = nt.tml; tml; tml = tml->next) {
872  Memb_list* ml = tml->ml;
873  mod_f_t s = corenrn.get_memb_func(tml->index).initialize;
874  if (s) {
875  (*s)(&nt, ml, tml->index);
876  }
877  }
878  }
879  _nrn_skip_initmodel = false;
880 
881  // if PatternStim exists, needs initialization
882  for (NrnThreadMembList* tml = nrn_threads[0].tml; tml; tml = tml->next) {
883  if (tml->index == patstimtype && patstim_index >= 0 && patstim_te > 0.0) {
884  Memb_list* ml = tml->ml;

886  /* below correct only for AoS */
887  0, ml->nodecount, ml->data, ml->pdata, ml->_thread,
888  nrn_threads, 0.0);
