202 // map GIDs of every compartment, it consist in a backward sweep then forward sweep algorithm
203 std::vector<int> ReportHandler::map_gids(const NrnThread& nt) const {
204  std::vector<int> nodes_gid(nt.end, -1);
205  // backward sweep: from presyn compartment propagate back GID to parent
206  for (int i = 0; i < nt.n_presyn; i++) {
207  const int gid = nt.presyns[i].gid_;
208  const int thvar_index = nt.presyns[i].thvar_index_;
209  // only for non artificial cells
210  if (thvar_index >= 0) {
211  // setting all roots gids of the presyns nodes,
212  // index 0 have parent set to 0, so we must stop at j > 0
213  // also 0 is the parent of all, so it is an error to attribute a GID to it.
214  nodes_gid[thvar_index] = gid;
215  for (int j = thvar_index; j > 0; j = nt._v_parent_index[j]) {
216  nodes_gid[nt._v_parent_index[j]] = gid;
217  }
218  }
219  }
220  // forward sweep: setting all compartements nodes to the GID of its root
221  // already sets on above loop. This is working only because compartments are stored in order
222  // parents follow by childrens
223  for (int i = nt.ncell + 1; i < nt.end; i++) {
224  nodes_gid[i] = nodes_gid[nt._v_parent_index[i]];
225  }
226  return nodes_gid;
227 }
