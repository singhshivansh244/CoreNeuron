43  * Split filter string ("mech.var_name") into mech_id and var_name
44  */
45 void parse_filter_string(char* filter, ReportConfiguration& config) {
46  char* token = strtok(filter, ".");
47  if (!token) {
48  std::cerr << "Error : Invalid report variable, should be mch_name.var_name" << std::endl;
49  abort();
50  }
51  strcpy(config.mech_name, token);
52  token = strtok(nullptr, "\n");
53  if (!token) {
54  std::cerr << "Error : Invalid report variable, should be mch_name.var_name" << std::endl;
55  abort();
56  }
57  strcpy(config.var_name, token);
58 }

61  const char* output_dir,
62  std::string& spikes_population_name) {

70 
71  FILE* fp = fopen(conf_file, "r");

74  << std::endl;
75  abort();
76  }
77 
78  fgets(raw_line, REPORT_MAX_FILEPATH_LEN, fp);
79  sscanf(raw_line, "%d\n", &num_reports);
80  for (int i = 0; i < num_reports; i++) {
81  reports.push_back(ReportConfiguration());
82  ReportConfiguration& report = reports[reports.size() - 1];
83  // mechansim id registered in coreneuron
84  report.mech_id = -1;
85  report.buffer_size = 4; // default size to 4 Mb
86  fgets(raw_line, REPORT_MAX_FILEPATH_LEN, fp);
87  sscanf(raw_line, "\n%s %s %s %s %s %s %d %lf %lf %lf %d %d %s\n", report.name,
88  report.target_name, report.type_str, report_on, report.unit, report.format, &is_soma,
89  &report.report_dt, &report.start, &report.stop, &report.num_gids,
90  &report.buffer_size, report.population_name);
91  for (int i = 0; i < REPORT_MAX_NAME_LEN; i++) {
92  report.type_str[i] = tolower(report.type_str[i]);
93  }
94  sprintf(report.output_path, "%s/%s", output_dir, report.name);
95  if (strcmp(report.type_str, "compartment") == 0) {
96  if (strcmp(report_on, "i_membrane") == 0) {
97  nrn_use_fast_imem = true;
98  report.type = IMembraneReport;
99  } else if (is_soma)
100  report.type = SomaReport;
101  else

103  } else if (strcmp(report.type_str, "synapse") == 0) {
104  report.type = SynapseReport;
105  } else {
106  std::cerr << "Report error: unsupported type " << report.type_str << "\n";
107  abort();
108  }
109 
110  if (report.type == SynapseReport)
111  parse_filter_string(report_on, report);
112 
113  if (report.num_gids) {
114  gids = (int*)calloc(report.num_gids, sizeof(int));
115  fread(gids, sizeof(int), report.num_gids, fp);
116  // extra new line
117  fgets(raw_line, REPORT_MAX_FILEPATH_LEN, fp);
118  for (int i = 0; i < report.num_gids; i++) {
119  report.target.insert(gids[i]);
120  }
121  free(gids);
122  }
123  }
124  fgets(raw_line, REPORT_MAX_NAME_LEN, fp);
125  sscanf(raw_line, "%s\n", spikes_population);
126  spikes_population_name = spikes_population;
127  fclose(fp);
128  return reports;
129 }
