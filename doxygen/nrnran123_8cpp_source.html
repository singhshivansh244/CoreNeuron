122 uint32_t nrnran123_ipick(nrnran123_State* s) {
123  uint32_t rval;

125  assert(which < 4);

145 /* at cost of a cached value we could compute two at a time. */
146 double nrnran123_normal(nrnran123_State* s) {
147  double w, x, y;
148  double u1, u2;
149  do {
150  u1 = nrnran123_dblpick(s);
151  u2 = nrnran123_dblpick(s);
152  u1 = 2. * u1 - 1.;
153  u2 = 2. * u2 - 1.;
154  w = (u1 * u1) + (u2 * u2);
155  } while (w > 1);
156 
157  y = sqrt((-2. * log(w)) / w);
158  x = u1 * y;
159  return x;
160 }

162 double nrnran123_uint2dbl(uint32_t u) {
163  /* 0 to 2^32-1 transforms to double value in open (0,1) interval */
164  /* min 2.3283064e-10 to max (1 - 2.3283064e-10) */
165  return ((double)u + 1.0) * SHIFT32;
