58 void nrnmpi_init(int nrnmpi_under_nrncontrol, int* pargc, char*** pargv) {
59  nrnmpi_use = true;
60  nrnmpi_under_nrncontrol_ = nrnmpi_under_nrncontrol;
61  if (nrnmpi_under_nrncontrol_) {
62 #if !ALWAYS_CALL_MPI_INIT
63  bool b = false;
64  /* this is not good. depends on mpirun adding at least one
65  arg that starts with -p4 but that probably is dependent
66  on mpich and the use of the ch_p4 device. We are trying to
67  work around the problem that MPI_Init may change the working
68  directory and so when not invoked under mpirun we would like to
69  NOT call MPI_Init.
70  */
71  for (int i = 0; i < *pargc; ++i) {
72  if (strncmp("-p4", (*pargv)[i], 3) == 0) {
73  b = true;
74  break;
75  }
76  if (strcmp("--mpi", (*pargv)[i]) == 0) {
77  b = true;
78  break;
79  }
80  }
81  if (nrnmusic) {
82  b = true;
83  }
84  if (!b) {
85  nrnmpi_use = false;
86  nrnmpi_under_nrncontrol_ = 0;
87  return;
88  }
89 #endif
90  int flag = 0;
91  MPI_Initialized(&flag);
92 
93  if (!flag) {
94 #if defined(_OPENMP)
95  int required = MPI_THREAD_FUNNELED;
96  int provided;
97  nrn_assert(MPI_Init_thread(pargc, pargv, required, &provided) == MPI_SUCCESS);
98 
99  nrn_assert(required <= provided);
100 #else
101  nrn_assert(MPI_Init(pargc, pargv) == MPI_SUCCESS);
102 #endif
103  }
104  }
105  grp_bbs = MPI_GROUP_NULL;
106  grp_net = MPI_GROUP_NULL;
107  nrn_assert(MPI_Comm_dup(MPI_COMM_WORLD, &nrnmpi_world_comm) == MPI_SUCCESS);
108  nrn_assert(MPI_Comm_dup(nrnmpi_world_comm, &nrnmpi_comm) == MPI_SUCCESS);
109  nrn_assert(MPI_Comm_dup(nrnmpi_world_comm, &nrn_bbs_comm) == MPI_SUCCESS);
110  nrn_assert(MPI_Comm_rank(nrnmpi_world_comm, &nrnmpi_myid_world) == MPI_SUCCESS);
111  nrn_assert(MPI_Comm_size(nrnmpi_world_comm, &nrnmpi_numprocs_world) == MPI_SUCCESS);
112  nrnmpi_numprocs = nrnmpi_numprocs_bbs = nrnmpi_numprocs_world;
113  nrnmpi_myid = nrnmpi_myid_bbs = nrnmpi_myid_world;
114  nrnmpi_spike_initialize();
115 
116  if (nrnmpi_myid == 0) {
117 #if defined(_OPENMP)
118  printf(" num_mpi=%d\n num_omp_thread=%d\n\n", nrnmpi_numprocs_world,
119  omp_get_max_threads());
120 #else
121  printf(" num_mpi=%d\n\n", nrnmpi_numprocs_world);
122 #endif
123  }
124 }

211 /**
212  * Return local mpi rank within a shared memory node
213  *
214  * When performing certain operations, we need to know the rank of mpi
215  * process on a given node. This function uses MPI 3 MPI_Comm_split_type
216  * function and MPI_COMM_TYPE_SHARED key to find out the local rank.
217  */
218 int nrnmpi_local_rank() {
219  int local_rank = 0;
220 #if NRNMPI
221  MPI_Comm local_comm;
222  MPI_Comm_split_type(MPI_COMM_WORLD, MPI_COMM_TYPE_SHARED, nrnmpi_myid_world, MPI_INFO_NULL, &local_comm);
223  MPI_Comm_rank(local_comm, &local_rank);
224  MPI_Comm_free(&local_comm);
225 #endif
226  return local_rank;
227 }
