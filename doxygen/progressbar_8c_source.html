101 /**
102  * Increment an existing progressbar by `value` steps.
103  * Additionally issues a redraw in case a certain time interval has elapsed (min: 1sec)
104  * Reasons for a larger interval are:
105  * - Stdout is not TTY
106  * - Respect BAR_DRAW_COUNT_MAX
107  */
108 void progressbar_update(progressbar* bar, unsigned long value, double t) {
109  bar->value = value;
110  bar->t = t;
111  int sim_time = difftime(time(NULL), bar->start);
112 
113  // If there is not enough time passed to redraw the progress bar return
114  if ((sim_time - bar->prev_t) < bar->draw_time_interval) {
115  return;
116  }
117 
118  progressbar_draw(bar);
119 
120  bar->drawn_count++;
121  bar->prev_t = sim_time;
122 
123  if (bar->drawn_count >= BAR_DRAW_COUNT_MAX || sim_time < 15) {
124  // Dont change the interval after the limit. Simulation should be over any moment and
125  // avoid the calc of draw_time_interval which could raise DIV/0
126  // Also, dont do it the first 15sec to avoid really bad estimates which could potentially
127  // delay a better estimate too far away in the future.
128  return;
129  }
130 
131  // Sample ETA to calculate the next interval until the redraw of the progressbar
132  int eta_s = progressbar_remaining_seconds(bar);
133  bar->draw_time_interval = eta_s / (BAR_DRAW_COUNT_MAX - bar->drawn_count);
134 
135  if (bar->draw_time_interval < BAR_DRAW_INTERVAL_NOTTY) {
136  bar->draw_time_interval =
137  isatty(STDOUT_FILENO)
138  ? ((bar->draw_time_interval < BAR_DRAW_INTERVAL) ? BAR_DRAW_INTERVAL
139  : bar->draw_time_interval)
140  : BAR_DRAW_INTERVAL_NOTTY;
141  }
142 }

182 static int progressbar_remaining_seconds(const progressbar* bar) {
183  double offset = difftime(time(NULL), bar->start);
184  if (bar->value > 0 && offset > 0) {
185  return (offset / (double)bar->value) * (bar->max - bar->value);
186  } else {

201 static void progressbar_draw(const progressbar* bar) {
202  int screen_width = get_screen_width();
203  int label_length = strlen(bar->label);
204  int bar_width = progressbar_bar_width(screen_width, label_length);
205  int label_width = progressbar_label_width(screen_width, label_length, bar_width);
206 
207  int progressbar_completed = (bar->value >= bar->max);
208  int bar_piece_count = bar_width - BAR_BORDER_WIDTH;
209  int bar_piece_current = (progressbar_completed)
210  ? bar_piece_count
211  : bar_piece_count * ((double)bar->value / bar->max);
212 
213  progressbar_time_components eta =
214  (progressbar_completed)
215  ? progressbar_calc_time_components(difftime(time(NULL), bar->start))
216  : progressbar_calc_time_components(progressbar_remaining_seconds(bar));
217 
218  if (label_width == 0) {
219  // The label would usually have a trailing space, but in the case that we don't print
220  // a label, the bar can use that space instead.
221  bar_width += 1;
222  } else {
223  // Draw the label
224  fwrite(bar->label, 1, label_width, stdout);
225  fputc(' ', stdout);
226  }
227 
228  // Draw the progressbar
229  fputc(bar->format.begin, stdout);
230  progressbar_write_char(stdout, bar->format.fill, bar_piece_current);
231  progressbar_write_char(stdout, ' ', bar_piece_count - bar_piece_current);
232  fputc(bar->format.end, stdout);
233 
234  // Draw the ETA
235  fputc(' ', stdout);
236  fprintf(stdout, ETA_FORMAT, bar->t, eta.hours, eta.minutes, eta.seconds);
237  fputc('\r', stdout);
238  fflush(stdout);
239 }
