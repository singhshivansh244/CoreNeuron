191 /*
192 The compressed spike format is restricted to the fixed step method and is
193 a sequence of unsigned char.
194 nspike = buf[0]*256 + buf[1]
195 a sequence of spiketime, localgid pairs. There are nspike of them.
196  spiketime is relative to the last transfer time in units of dt.
197  note that this requires a mindelay < 256*dt.
198  localgid is an unsigned int, unsigned short,
199  or unsigned char in size depending on the range and thus takes
200  4, 2, or 1 byte respectively. To be machine independent we do our
201  own byte coding. When the localgid range is smaller than the true
202  gid range, the gid->PreSyn are remapped into
203  hostid specific maps. If there are not many holes, i.e just about every
204  spike from a source machine is delivered to some cell on a
205  target machine, then instead of a hash map, a vector is used.
206 The allgather sends the first part of the buf and the allgatherv buffer
207 sends any overflow.
208 */
209 int nrnmpi_spike_exchange_compressed() {
210  if (!displs) {
211  np = nrnmpi_numprocs;
212  displs = (int*)emalloc(np * sizeof(int));
213  displs[0] = 0;
214  byteovfl = (int*)emalloc(np * sizeof(int));
215  }
216 
217  MPI_Allgather(spfixout_, ag_send_size_, MPI_BYTE, spfixin_, ag_send_size_, MPI_BYTE,
218  nrnmpi_comm);
219  int novfl = 0;
220  int ntot = 0;
221  int bstot = 0;
222  for (int i = 0; i < np; ++i) {
223  displs[i] = bstot;
224  int idx = i * ag_send_size_;
225  int n = spfixin_[idx++] * 256;
226  n += spfixin_[idx++];
227  ntot += n;
228  nin_[i] = n;
229  if (n > ag_send_nspike_) {
230  int bs = 2 + n * (1 + localgid_size_) - ag_send_size_;
231  byteovfl[i] = bs;
232  bstot += bs;
233  novfl += n - ag_send_nspike_;
234  } else {
235  byteovfl[i] = 0;
236  }
237  }
238  if (novfl) {
239  if (ovfl_capacity_ < novfl) {
240  ovfl_capacity_ = novfl + 10;
241  free(spfixin_ovfl_);
242  spfixin_ovfl_ = (unsigned char*)emalloc(ovfl_capacity_ * (1 + localgid_size_) *
243  sizeof(unsigned char));
244  }
245  int bs = byteovfl[nrnmpi_myid];
246  /*
247  note that the spfixout_ buffer is one since the overflow
248  is contiguous to the first part. But the spfixin_ovfl_ is
249  completely separate from the spfixin_ since the latter
250  dynamically changes its size during a run.
251  */
252  MPI_Allgatherv(spfixout_ + ag_send_size_, bs, MPI_BYTE, spfixin_ovfl_, byteovfl, displs,
253  MPI_BYTE, nrnmpi_comm);
254  }
255  ovfl_ = novfl;
256  return ntot;
257 }

368 static void pgvts_op(double* in, double* inout, int* len, MPI_Datatype* dptr) {
369  int r = 0;
370  if (*dptr != MPI_DOUBLE)
371  printf("ERROR in mpispike.c! *dptr should be MPI_DOUBLE.");
372  if (*len != 4)
373  printf("ERROR in mpispike.c! *len should be 4.");
374  if (in[0] < inout[0]) {
375  /* least time has highest priority */
376  r = 1;
377  } else if (in[0] == inout[0]) {
378  /* when times are equal then */
379  if (in[1] < inout[1]) {
380  /* NetParEvent done last */
381  r = 1;
382  } else if (in[1] == inout[1]) {
383  /* when times and ops are equal then */
384  if (in[2] < inout[2]) {
385  /* init done next to last.*/
386  r = 1;
387  } else if (in[2] == inout[2]) {
388  /* when times, ops, and inits are equal then */
389  if (in[3] < inout[3]) {
390  /* choose lowest rank */
391  r = 1;
392  }
393  }
394  }
395  }
396  if (r) {
397  for (int i = 0; i < 4; ++i) {
398  inout[i] = in[i];
399  }
400  }
401 }

403 int nrnmpi_pgvts_least(double* tt, int* op, int* init) {
404  double ibuf[4], obuf[4];
405  ibuf[0] = *tt;
406  ibuf[1] = (double)(*op);
407  ibuf[2] = (double)(*init);
408  ibuf[3] = (double)nrnmpi_myid;
409  std::memcpy(obuf, ibuf, 4 * sizeof(double));
410 
411  MPI_Allreduce(ibuf, obuf, 4, MPI_DOUBLE, mpi_pgvts_op, nrnmpi_comm);
412  assert(obuf[0] <= *tt);
413  if (obuf[0] == *tt) {
414  assert((int)obuf[1] <= *op);
415  if ((int)obuf[1] == *op) {
416  assert((int)obuf[2] <= *init);
417  if ((int)obuf[2] == *init) {
418  assert((int)obuf[3] <= nrnmpi_myid);
419  }
420  }
421  }
422  *tt = obuf[0];
423  *op = (int)obuf[1];
424  *init = (int)obuf[2];
425  if (nrnmpi_myid == (int)obuf[3]) {
426  return 1;
427  }
428  return 0;
429 }
