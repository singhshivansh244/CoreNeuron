66  * may need to implement a special copy-construtor. This is documented here:
67  * \link: https://devblogs.nvidia.com/unified-memory-in-cuda-6/

126 /** Independent function to compute the needed chunkding,
127  the chunk argument is the number of doubles the chunk is chunkded upon.
128 */
129 template <int chunk>
130 inline int soa_padded_size(int cnt, int layout) {
131  int imod = cnt % chunk;
132  if (layout == 1)
133  return cnt;
134  if (imod) {
135  int idiv = cnt / chunk;
136  return (idiv + 1) * chunk;
137  }
138  return cnt;
139 }
