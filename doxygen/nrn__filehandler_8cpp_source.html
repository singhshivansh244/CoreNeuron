44 void FileHandler::open(const std::string& filename, std::ios::openmode mode) {
45  nrn_assert((mode & (std::ios::in | std::ios::out)));
46  close();
47  F.open(filename, mode | std::ios::binary);
48  if (!F.is_open()) {
49  std::cerr << "cannot open file '" << filename << "'" << std::endl;
50  }
51  nrn_assert(F.is_open());
52  current_mode = mode;
53  char version[256];
54  if (current_mode & std::ios::in) {
55  F.getline(version, sizeof(version));
56  nrn_assert(!F.fail());
57  check_bbcore_write_version(version);
58  }
59  if (current_mode & std::ios::out) {
60  F << bbcore_write_version << "\n";
61  }
62 }

104 void FileHandler::read_checkpoint_assert() {
105  char line_buf[max_line_length];
106 
107  F.getline(line_buf, sizeof(line_buf));
108  nrn_assert(!F.fail());
109 
110  int i;
111  int n_scan = sscanf(line_buf, "chkpnt %d\n", &i);
112  if (n_scan != 1) {
113  fprintf(stderr, "no chkpnt line for %d\n", chkpnt);
114  }
115  nrn_assert(n_scan == 1);
116  if (i != chkpnt) {
117  fprintf(stderr, "file chkpnt %d != expected %d\n", i, chkpnt);
118  }
119  nrn_assert(i == chkpnt);
120  ++chkpnt;
121 }
